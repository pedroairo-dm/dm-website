function DisableTabSectionFields(TAB_ID, SECTION_IDX)
{
//TAB_ID --> tab0 for the first TAB
var tab = document.getElementById(TAB_ID);
//SECTION_IDX --> 0 for the first section
var sec = tab.childNodes.rows;
var elems = sec.getElementsByTagName('*');
var i=0;
for(i=0; i < elems.length; ++i)
{
if (elems.Disabled != "undefined")
elems.Disabled=true;
}
}
